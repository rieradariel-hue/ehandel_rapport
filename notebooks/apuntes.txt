def calculate_order_distribution(df):
    """
    Analyserar spridning av ordervärden
    Returnerar dict med nyckeltal för distribution
    """
    return {
        'medelvarde': df['revenue'].mean(),
        'median': df['revenue'].median(),
        'standardavvikelse': df['revenue'].std(),
        'percentil_25': df['revenue'].quantile(0.25),
        'percentil_75': df['revenue'].quantile(0.75),
        'min': df['revenue'].min(),
        'max': df['revenue'].max()
    }
